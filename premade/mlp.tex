% NEURAL NETWORK activation
% https://www.youtube.com/watch?v=aircAruvnKk&list=PLZHQObOWTQDNU6R1_67000Dx_ZCJB-3pi&index=1

\begin{tikzpicture}[x=2.7cm,y=1.6cm]
  \message{^^JNeural network activation}
  \def\NI{4} % number of nodes in input layers
  \def\NH{5} % number of nodes in output layers
  \def\NO{3} % number of nodes in output layers  
  \def\yshift{0.8} % shift last node for dots
  
  % INPUT LAYER
  \foreach \i [evaluate={\c=int(\i==\NI); \y=\NI/2-\i-\c*\yshift; \index=(\i<\NI?int(\i):"p");}]
              in {1,...,\NI}{ % loop over nodes
    \node[node in,outer sep=0.6] (NI-\i) at (0,\y) {\:input\:};
  }
  
  % HIDDEN LAYER
  \foreach \i [evaluate={\c=int(\i==\NH); \y=\NH/2-\i-\c*\yshift; \index=(\i<\NH?int(\i):"m");}]
    in {\NH,...,1}{ % loop over nodes
      \node[node hidden]
        (NH-\i) at (1.5,\y) {\:hidden\:};
      \foreach \j [evaluate={\index=(\j<\NI?int(\j):"p");}] in {1,...,\NI}{ % loop over nodes in previous layer
        \draw[connect,mygrey,line width=0.1,fill=mygrey] (NI-\j) -- (NH-\i);
        \draw[connect,mygrey,fill=mygrey] (NI-\j) -- (NH-\i)
          node[pos=0.50] {};
      }
  }
  
  % OUTPUT LAYER
  \foreach \i [evaluate={\c=int(\i==\NO); \y=\NO/2-\i-\c*\yshift; \index=(\i<\NO?int(\i):"m");}]
    in {\NO,...,1}{ % loop over nodes
      \node[node out]
        (NO-\i) at (3,\y) {\:output\:};
      \foreach \j [evaluate={\index=(\j<\NH?int(\j):"p");}] in {1,...,\NH}{ % loop over nodes in previous layer
        \draw[connect,mygrey,line width=.1] (NH-\j) -- (NO-\i);
        \draw[connect,mygrey] (NH-\j) -- (NO-\i)
          node[pos=0.50] {};
      }
  }
  
  % DOTS
  \path (NI-\NI) --++ (0,1+\yshift) node[midway,scale=1.2] {$\vdots$};
  \path (NH-\NH) --++ (0,1+\yshift) node[midway,scale=1.2] {$\vdots$};
  \path (NO-\NO) --++ (0,1+\yshift) node[midway,scale=1.2] {$\vdots$};  
  
\end{tikzpicture}

\end{document}
